# Date to Datetime format
df_monthly['date'] = pd.to_datetime(df_monthly['date'])
df_yearly['date'] = pd.to_datetime(df_yearly['date'])

# extract data from 2000 
df_monthly = df_monthly[df_monthly['date'].dt.year >= 2000]
df_yearly = df_yearly[df_yearly['date'].dt.year >= 2000]

# year column for merging two files
df_monthly['year'] = df_monthly['date'].dt.year
df_yearly['year'] = df_yearly['date'].dt.year

# merging two files 
df_merged = df_monthly.merge(df_yearly, on = ['year', 'area'], how = 'left')
df_merged.to_csv('merged_housing_data_2000_latest.csv', index = False)

# X (independent variables), Y (dependent variables)
# split dataset  into training (80%) and testing (20%)
X = df_cleaned[selected_features]
Y = df_cleaned['average_price']
X_train, X_test, Y_train, Y_test = train_test_split(X, Y, test_size=0.2, random_state=42)

# ----- linear regression model -----
# training by X_train, Y_train
lr_model = LinearRegression()
lr_model.fit(X_train, Y_train)

# prediction of Y when using X_test values 
lr_prediction_Y = lr_model.predict(X_test)

# evaluation
lr_metrics = {
    'Model': 'Linear Regression',
    'MAE': mean_absolute_error(Y_test, lr_prediction_Y), # Y_test VS Y prediction
    'MSE': mean_squared_error(Y_test, lr_prediction_Y),
    'R2 Score': r2_score(Y_test, lr_prediction_Y)
}
print(lr_metrics)

# save predictions
df_lr_predictions = pd.DataFrame({
    'Actual Price': Y_test.values,
    'Predicted Price': lr_prediction_Y
})
df_lr_predictions.to_csv('predicted_prices_lr.csv', index=False)


# visualisation 
# data of actual and predicted price
plt.figure(figsize=(8, 5))
plt.scatter(df_lr_predictions['Actual Price'], df_lr_predictions['Predicted Price'], 
            alpha=0.5, color='darkseagreen', label='Linear Regression Predictions')

# line 
min_val = min(df_lr_predictions['Actual Price'].min(), df_lr_predictions['Predicted Price'].min())
max_val = max(df_lr_predictions['Actual Price'].max(), df_lr_predictions['Predicted Price'].max())
plt.plot([min_val, max_val], [min_val, max_val], color='red', linewidth=3)

plt.xlabel('Actual Price')
plt.ylabel('Predicted Price')
plt.title('Actual VS Predicted Prices (Linear Regression)')
plt.legend()
plt.grid(True)
plt.show()
